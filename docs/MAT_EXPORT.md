func reorderCToFortran<T>(_ arr: [T], dims: (Int, Int, Int)) -> [T] {
    let (d0, d1, d2) = dims
    var result = arr
    
    for i0 in 0..<d0 {
        for i1 in 0..<d1 {
            for i2 in 0..<d2 {
                let cIndex = i0 * (d1 * d2) + i1 * d2 + i2
                let fortranIndex = i0 + d0 * (i1 + d1 * i2)

        }
    }

}
